function(check_env_defined name)
    if(DEFINED ENV{${name}})
        MESSAGE(STATUS "Environment variable ${name} set to: $ENV{${name}}")
    else()
        MESSAGE(SEND_ERROR "Environment variable ${name} not defined!")
    endif()
endfunction()
